FROM ubuntu:24.04

ENV OPTAPT=""
#ENV OPTAPT="-o Acquire::AllowInsecureRepositories=true -o Acquire::AllowDowngradeToInsecureRepositories=true"
RUN \
  apt-get ${OPTAPT} -y update && \
  apt-get ${OPTAPT} -y upgrade && \
  apt-get ${OPTAPT} -y install gcc-riscv64-linux-gnu qemu-user && \
  apt-get ${OPTAPT} -y install ghc cabal-install && \
  apt-get ${OPTAPT} -y install swi-prolog && \
  apt-get ${OPTAPT} -y install ruby-full && \
  apt-get ${OPTAPT} -y install guile-3.0 && \
  apt-get ${OPTAPT} -y install nodejs npm && \
  apt-get ${OPTAPT} -y install python3 python3-venv && \
  rm -rf /var/lib/apt/lists/*

ENV DIRRSC=/env/risc-v
RUN mkdir -p ${DIRRSC}
COPY --chmod=644 exersc.sh ${DIRRSC}

ENV DIRNOD=/env/nodejs
RUN mkdir -p ${DIRNOD} && \
  cd ${DIRNOD} && \
  npm init -y && \
  npm install typescript
COPY --chmod=644 trntys.js ${DIRNOD}
COPY --chmod=644 exetys.sh ${DIRNOD}

ENV DIRPYT=/env/python
RUN python3 -m venv ${DIRPYT}
RUN ARC=$(uname -m) && \
  if test "${ARC}" = 'x86_64'; \
  then \
    ${DIRPYT}/bin/pip3 install --no-cache-dir torch --index-url https://download.pytorch.org/whl/cpu; \
  else \
    ${DIRPYT}/bin/pip3 install --no-cache-dir torch; \
  fi
RUN ${DIRPYT}/bin/pip3 install --no-cache-dir aiohttp
